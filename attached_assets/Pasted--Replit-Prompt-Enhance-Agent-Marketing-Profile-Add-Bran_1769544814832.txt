## ğŸ”Œ API Update

Update your existing settings PATCH endpoint to handle the new fields:

```typescript
// In your existing settings route
app.patch('/api/settings', async (req, res) => {
  try {
    const userId = req.user?.id;
    if (!userId) return res.status(401).json({ error: 'Unauthorized' });

    const {
      // Existing fields
      firstName,
      lastName,
      email,
      phone,
      title,
      company,
      profilePhoto,
      
      // NEW branding fields
      qrCode,
      companyLogo,
      companyLogoUseDefault,
      secondaryLogo,
      secondaryLogoUseDefault,
    } = req.body;

    const [updated] = await db
      .update(users)
      .set({
        // Existing fields
        firstName,
        lastName,
        email,
        phone,
        title,
        company,
        profilePhoto,
        
        // NEW branding fields
        qrCode,
        companyLogo,
        companyLogoUseDefault,
        secondaryLogo,
        secondaryLogoUseDefault,
        
        updatedAt: new Date(),
      })
      .where(eq(users.id, userId))
      .returning();

    res.json(updated);
  } catch (error) {
    console.error('Settings update error:', error);
    res.status(500).json({ error: 'Failed to update settings' });
  }
});
```

app.get('/api/settings', async (req, res) => {
