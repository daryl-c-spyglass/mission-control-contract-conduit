```tsx
  slideNumber: number;
  totalSlides: number;

}

const LRE_LOGO_PATH = '/logos/LRE_SGR_White.png';

// For Preview (React component)
export function SlideFooter({ 
  slideNumber, 
  totalSlides, 
  showLogo = slideNumber > 1 // Show logo on slides 2-44

  return (
    <div className="flex justify-between items-center px-6 py-4 border-t border-gray-100">
      {/* Left: Logo (slides 2-44) or Text (slide 1) */}
      <div className="flex items-center">
        {showLogo ? (
          <img 
            src={LRE_LOGO_PATH}
            alt="Spyglass Realty"
```

### Option A: Shared Slide Data

```typescript
// File: lib/slide-content.ts

export interface SlideContent {
  id: string;
  title: string;
  type: 'cover' | 'video' | 'content' | 'comps' | 'property';
  showFooterLogo: boolean;
  content: any;
}

export function getSlideContent(slideId: string, slideNumber: number): SlideContent {
  return {
    id: slideId,
    title: getSlideTitleById(slideId),
    type: getSlideTypeById(slideId),
    showFooterLogo: slideNumber > 1, // Logo on slides 2-44
    content: getSlideContentById(slideId),
  };
}
```
