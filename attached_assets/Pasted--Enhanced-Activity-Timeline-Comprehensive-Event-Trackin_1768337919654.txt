```typescript
// shared/schema.ts
export const timelineEvents = pgTable("timeline_events", {
  id: serial("id").primaryKey(),
  transactionId: integer("transaction_id").references(() => transactions.id).notNull(),
  userId: varchar("user_id").references(() => users.id),
  eventType: varchar("event_type", { length: 50 }).notNull(),
  eventCategory: varchar("event_category", { length: 30 }).notNull(),
  title: text("title").notNull(),
  description: text("description"),
  metadata: json("metadata").$type<Record<string, any>>(),
  createdAt: timestamp("created_at").defaultNow().notNull(),
});

export type TimelineEvent = typeof timelineEvents.$inferSelect;
export type InsertTimelineEvent = typeof timelineEvents.$inferInsert;
```

### 3. Timeline Service (Backend)

```typescript
// server/services/timeline.ts
import { storage } from '../storage';
import { InsertTimelineEvent } from '@shared/schema';

export async function logTimelineEvent(
  transactionId: number,
  eventType: string,
  title: string,
  options?: {
    userId?: string;
    description?: string;
    metadata?: Record<string, any>;
  }
) {
  const event: InsertTimelineEvent = {
    transactionId,
    eventType,
    eventCategory: getCategoryFromType(eventType),
    title,
    userId: options?.userId,
    description: options?.description,
    metadata: options?.metadata,
  };

  return await storage.createTimelineEvent(event);
}

function getCategoryFromType(eventType: string): string {
  if (eventType.startsWith('transaction_') || eventType === 'status_changed') return 'transaction';
  if (eventType.startsWith('mls_') || eventType === 'price_changed' || eventType === 'photos_updated') return 'mls';
  if (eventType.startsWith('document_')) return 'documents';
  if (eventType.includes('graphic') || eventType.includes('flyer') || eventType.includes('asset') || eventType === 'quick_create_all') return 'marketing';
  if (eventType.startsWith('cma_')) return 'cma';
  if (eventType.startsWith('coordinator_') || eventType === 'note_added') return 'team';
  if (eventType.includes('date') || eventType === 'deadline_approaching') return 'dates';
  if (eventType === 'email_sent' || eventType === 'slack_notification') return 'communication';
  return 'other';
}
```

```tsx
// components/timeline-tab.tsx
import { useQuery } from '@tanstack/react-query';
import { Card, CardContent } from '@/components/ui/card';
import { Badge } from '@/components/ui/badge';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
import { Loader2 } from 'lucide-react';
import { useState } from 'react';
import { cn } from '@/lib/utils';
import { getEventConfig, TIMELINE_EVENT_CONFIG } from '@/lib/timeline-config';

interface TimelineTabProps {
  transactionId: number;
}

const CATEGORY_FILTERS = [
  { value: 'all', label: 'All Activities' },
  { value: 'transaction', label: 'Transaction' },
  { value: 'mls', label: 'MLS Data' },
  { value: 'documents', label: 'Documents' },
  { value: 'marketing', label: 'Marketing' },
  { value: 'cma', label: 'CMA' },
  { value: 'team', label: 'Team' },
  { value: 'dates', label: 'Dates' },
];

export function TimelineTab({ transactionId }: TimelineTabProps) {
  const [categoryFilter, setCategoryFilter] = useState('all');

  const { data: events = [], isLoading } = useQuery({
    queryKey: ['timeline', transactionId, categoryFilter],
    queryFn: async () => {
      const params = new URLSearchParams();
      if (categoryFilter !== 'all') {
        params.set('category', categoryFilter);
      }
      const res = await fetch(`/api/transactions/${transactionId}/timeline?${params}`);
      if (!res.ok) throw new Error('Failed to fetch timeline');
      return res.json();
    },
  });

  // Group events by date
  const groupedEvents = groupEventsByDate(events);

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-64">
        <Loader2 className="h-8 w-8 animate-spin text-muted-foreground" />
      </div>
    );
  }

  return (
    <div className="space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <h2 className="text-xl font-semibold">Activity Timeline</h2>
        <Select value={categoryFilter} onValueChange={setCategoryFilter}>
          <SelectTrigger className="w-40">
            <SelectValue />
          </SelectTrigger>
          <SelectContent>
            {CATEGORY_FILTERS.map((filter) => (
              <SelectItem key={filter.value} value={filter.value}>
                {filter.label}
              </SelectItem>
            ))}
          </SelectContent>
        </Select>
      </div>

      {/* Timeline */}
      {events.length === 0 ? (
        <Card>
          <CardContent className="py-12 text-center text-muted-foreground">
            No activity recorded yet
          </CardContent>
        </Card>
      ) : (
        <div className="space-y-8">
          {Object.entries(groupedEvents).map(([date, dateEvents]) => (
            <div key={date}>
              {/* Date Header */}
              <div className="flex items-center gap-3 mb-4">
                <div className="h-px flex-1 bg-border" />
                <span className="text-sm font-medium text-muted-foreground">{date}</span>
                <div className="h-px flex-1 bg-border" />
              </div>

              {/* Events for this date */}
              <div className="space-y-3">
                {(dateEvents as any[]).map((event) => (
                  <TimelineEventCard key={event.id} event={event} />
                ))}
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

function TimelineEventCard({ event }: { event: any }) {
  const config = getEventConfig(event.eventType);

  return (
    <div className="flex gap-4 items-start">
      {/* Emoji indicator */}
      <div className={cn(
        "flex-shrink-0 w-10 h-10 rounded-full flex items-center justify-center text-lg",
        config.color.replace('bg-', 'bg-opacity-20 bg-')
      )}>
        {config.emoji}
      </div>

      {/* Content */}
      <div className="flex-1 min-w-0">
        <div className="flex items-center gap-2 flex-wrap">
          <p className="font-medium text-sm">{event.title}</p>
          <Badge variant="secondary" className="text-xs">
            {formatCategory(event.eventCategory)}
          </Badge>
        </div>
        {event.description && (
          <p className="text-sm text-muted-foreground mt-1">{event.description}</p>
        )}
        <p className="text-xs text-muted-foreground mt-1">
          {formatTime(event.createdAt)}
        </p>
      </div>
    </div>
  );
}

// Helper functions
function groupEventsByDate(events: any[]): Record<string, any[]> {
  const groups: Record<string, any[]> = {};
  
  events.forEach((event) => {
    const date = new Date(event.createdAt).toLocaleDateString('en-US', {
      weekday: 'long',
      month: 'long',
      day: 'numeric',
      year: 'numeric',
    });
    
    if (!groups[date]) {
      groups[date] = [];
    }
    groups[date].push(event);
  });
  
  return groups;
}

function formatTime(dateString: string): string {
  return new Date(dateString).toLocaleTimeString('en-US', {
    hour: 'numeric',
    minute: '2-digit',
    hour12: true,
  });
}

function formatCategory(category: string): string {
  return category.charAt(0).toUpperCase() + category.slice(1);
}
```

```typescript
// server/routes.ts
app.get('/api/transactions/:id/timeline', async (req, res) => {
  try {
    const transactionId = parseInt(req.params.id);
    const category = req.query.category as string | undefined;
    
    let events = await storage.getTimelineEvents(transactionId);
    
    // Filter by category if specified
    if (category && category !== 'all') {
      events = events.filter(e => e.eventCategory === category);
    }
    
    // Sort by newest first
    events.sort((a, b) => new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime());
    
    res.json(events);
  } catch (error) {
    res.status(500).json({ error: 'Failed to fetch timeline' });
  }
});
```

```typescript
// server/storage.ts
async getTimelineEvents(transactionId: number): Promise<TimelineEvent[]> {
  return await this.db.select()
    .from(timelineEvents)
    .where(eq(timelineEvents.transactionId, transactionId))
    .orderBy(desc(timelineEvents.createdAt));
}

async createTimelineEvent(event: InsertTimelineEvent): Promise<TimelineEvent> {
  const result = await this.db.insert(timelineEvents).values(event).returning();
  return result[0];
}
```

## MLS Sync Deduplication

To prevent duplicate "MLS synced" entries, only log when there are actual changes:

```typescript
// In MLS sync function
async function syncMLSData(transactionId: number, mlsNumber: string) {
  const currentData = await storage.getMLSData(transactionId);
  const newData = await fetchFromRepliers(mlsNumber);
  
  // Check for meaningful changes
  const hasChanges = 
    currentData?.photoCount !== newData.images?.length ||
    currentData?.listPrice !== newData.listPrice ||
    currentData?.status !== newData.status;
  
  if (hasChanges) {
    // Update data
    await storage.updateMLSData(transactionId, newData);
    
    // Log specific changes
    if (currentData?.listPrice !== newData.listPrice) {
      await TimelineLogger.priceChanged(transactionId, currentData?.listPrice || 0, newData.listPrice);
    }
    
    if (currentData?.status !== newData.status) {
      await logTimelineEvent(transactionId, 'mls_status_changed', 
        `MLS status changed to ${newData.status}`);
    }
    
    // Only log generic sync if nothing specific changed
    if (currentData?.listPrice === newData.listPrice && currentData?.status === newData.status) {
      await TimelineLogger.mlsSynced(transactionId, newData.images?.length || 0, newData.comparables?.length || 0);
    }
  }
  // Don't log anything if no changes
}
```
