  slideNumber: number;
  totalSlides: number;
}

const SPYGLASS_LOGO = '/logos/SpyglassRealty_Logo_White.png';

export function SlideFooter({ slideNumber, totalSlides }: SlideFooterProps) {
  // Slide 1 uses text, Slides 2+ use logo
  const showLogo = slideNumber > 1;
  
  return (
    <div className="flex justify-between items-center px-6 py-3 border-t border-gray-100">
      {/* Left: Logo (slides 2-44) or Text (slide 1) */}
      <div className="flex items-center h-8">
        {showLogo ? (
          <img 
            src={SPYGLASS_LOGO}
            alt="Spyglass Realty"
            className="h-6 w-auto rounded"
          />
        ) : (
          <span className="text-sm text-gray-500">Spyglass Realty</span>
        )}
      </div>
      
      {/* Right: Slide number */}
      <span className="text-sm text-gray-400">
        Slide {slideNumber} of {totalSlides}
      </span>
    </div>
  );
}
