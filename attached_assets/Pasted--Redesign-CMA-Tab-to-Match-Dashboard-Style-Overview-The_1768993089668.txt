# Redesign CMA Tab to Match Dashboard Style

## Overview
The CMA tab looks inconsistent with the Dashboard. Redesign it to match the clean, professional style while organizing action buttons into dropdown menus.

---

## Design Goals
- Match Dashboard's light, clean aesthetic
- White cards with subtle borders
- Muted, consistent status colors
- Dropdown menus for action buttons
- Default view: Grid
- Yellow banner only when there are unsaved changes
- Map theme syncs with light/dark mode

---

## Part 2: Status Colors (Unified)

Create a consistent color system for property statuses:

```tsx
// lib/statusColors.ts

export const getStatusConfig = (status: string) => {
  const normalized = status?.toLowerCase().replace(/\s+/g, '') || '';
  
  if (normalized.includes('active') && !normalized.includes('under')) {
    return {
      label: 'Active',
      badge: 'bg-green-500 text-white',
      dot: 'bg-green-500',
      marker: '#22c55e',
    };
  }
  
  if (normalized.includes('closed') || normalized.includes('sold')) {
    return {
      label: 'Closed',
      badge: 'bg-gray-500 text-white',
      dot: 'bg-gray-500',
      marker: '#6b7280',
    };
  }
  
  if (normalized.includes('under') || normalized.includes('contract')) {
    return {
      label: 'Under Contract',
      badge: 'bg-amber-500 text-white',
      dot: 'bg-amber-500',
      marker: '#f59e0b',
    };
  }
  
  if (normalized.includes('pending')) {
    return {
      label: 'Pending',
      badge: 'bg-yellow-500 text-white',
      dot: 'bg-yellow-500',
      marker: '#eab308',
    };
  }
  
  return {
    label: status || 'Unknown',
    badge: 'bg-gray-500 text-white',
    dot: 'bg-gray-500',
    marker: '#6b7280',
  };
};
```

```tsx
// components/ui/StatusBadge.tsx

import { cn } from '@/lib/utils';
import { getStatusConfig } from '@/lib/statusColors';

interface StatusBadgeProps {
  status: string;
  size?: 'sm' | 'md';
}

export function StatusBadge({ status, size = 'md' }: StatusBadgeProps) {
  const config = getStatusConfig(status);
  
  return (
    <span className={cn(
      "inline-flex items-center font-medium rounded-full whitespace-nowrap",
      size === 'sm' ? "px-2 py-0.5 text-xs" : "px-2.5 py-1 text-sm",
      config.badge
    )}>
      {config.label}
    </span>
  );
}
```

## Part 7: Map View with Theme Sync

```tsx
// components/cma/CMAMapView.tsx

import { useEffect, useRef, useMemo } from 'react';
import Map, { Marker, NavigationControl } from 'react-map-gl';
import { useTheme } from 'next-themes'; // or your theme hook
import { Home } from 'lucide-react';
import { getStatusConfig } from '@/lib/statusColors';
import { formatCompactCurrency } from '@/lib/utils';
import 'mapbox-gl/dist/mapbox-gl.css';

interface CMAMapViewProps {
  comparables: any[];
  subject: any;
}

export function CMAMapView({ comparables, subject }: CMAMapViewProps) {
  const { theme, systemTheme } = useTheme();
  const currentTheme = theme === 'system' ? systemTheme : theme;
  
  // Map style based on theme
  const mapStyle = useMemo(() => {
    return currentTheme === 'dark' 
      ? 'mapbox://styles/mapbox/dark-v11'
      : 'mapbox://styles/mapbox/light-v11';
  }, [currentTheme]);

  // Calculate bounds to fit all markers
  const bounds = useMemo(() => {
    const allProperties = [subject, ...comparables].filter(p => p?.latitude && p?.longitude);
    if (allProperties.length === 0) return null;
    
    const lngs = allProperties.map(p => p.longitude);
    const lats = allProperties.map(p => p.latitude);
    
    return {
      longitude: (Math.min(...lngs) + Math.max(...lngs)) / 2,
      latitude: (Math.min(...lats) + Math.max(...lats)) / 2,
      zoom: 11,
    };
  }, [subject, comparables]);

  return (
    <div className="h-[500px] rounded-xl overflow-hidden border border-gray-200 dark:border-gray-700">
      <Map
        mapboxAccessToken={import.meta.env.VITE_MAPBOX_TOKEN}
        initialViewState={bounds || {
          longitude: -97.7431,
          latitude: 30.2672,
          zoom: 11
        }}
        style={{ width: '100%', height: '100%' }}
        mapStyle={mapStyle}
      >
        <NavigationControl position="top-right" />
        
        {/* Subject Property Marker */}
        {subject?.latitude && subject?.longitude && (
          <Marker 
            longitude={subject.longitude} 
            latitude={subject.latitude}
            anchor="center"
          >
            <div className="relative">
              <div className="w-10 h-10 bg-primary rounded-full border-3 border-white shadow-lg flex items-center justify-center">
                <Home className="w-5 h-5 text-white" />
              </div>
              <div className="absolute -bottom-1 left-1/2 -translate-x-1/2 w-0 h-0 border-l-4 border-r-4 border-t-4 border-transparent border-t-primary" />
            </div>
          </Marker>
        )}

        {/* Comparable Property Markers */}
        {comparables.map((property) => {
          if (!property.latitude || !property.longitude) return null;
          const config = getStatusConfig(property.standardStatus);
          
          return (
            <Marker
              key={property.id || property.listingId}
              longitude={property.longitude}
              latitude={property.latitude}
              anchor="center"
            >
              <div 
                className="px-2.5 py-1 rounded-full text-white text-xs font-semibold shadow-lg border-2 border-white cursor-pointer hover:scale-110 transition-transform"
                style={{ backgroundColor: config.marker }}
              >
                {formatCompactCurrency(property.closePrice || property.listPrice)}
              </div>
            </Marker>
          );
        })}

        {/* Legend */}
        <div className="absolute bottom-4 left-4 bg-white dark:bg-gray-900 rounded-xl shadow-lg p-4 border border-gray-200 dark:border-gray-700">
          <p className="text-xs font-semibold text-gray-700 dark:text-gray-300 uppercase tracking-wide mb-2">
            Legend
          </p>
          <div className="space-y-2">
            <div className="flex items-center gap-2">
              <div className="w-6 h-6 rounded-full bg-primary flex items-center justify-center">
                <Home className="w-3 h-3 text-white" />
              </div>
              <span className="text-sm text-gray-600 dark:text-gray-400">Subject Property</span>
            </div>
```

## Part 8: Utility Functions

```tsx
// lib/utils.ts - Add these helper functions

export function formatCurrency(value: number | undefined | null): string {
  if (value === undefined || value === null) return '—';
  return new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency: 'USD',
    minimumFractionDigits: 0,
    maximumFractionDigits: 0,
  }).format(value);
}

export function formatCompactCurrency(value: number | undefined | null): string {
  if (value === undefined || value === null) return '—';
  if (value >= 1000000) {
    return `$${(value / 1000000).toFixed(1)}M`;
  }
  if (value >= 1000) {
    return `$${(value / 1000).toFixed(0)}K`;
  }
  return `$${value}`;
}
```
