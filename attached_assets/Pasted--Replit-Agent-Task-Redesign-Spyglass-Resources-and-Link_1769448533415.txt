### Resource Item Component

```tsx
interface ResourceItemProps {
  resource: Resource;
  index: number;
  onToggle: (id: string) => void;
  onDelete: (id: string) => void;
  onEdit: (resource: Resource) => void;
}

function ResourceItem({ resource, index, onToggle, onDelete, onEdit }: ResourceItemProps) {
  return (
    <Draggable draggableId={resource.id} index={index}>
      {(provided, snapshot) => (
        <div 
          ref={provided.innerRef}
          {...provided.draggableProps}
          className={`flex items-center gap-3 px-4 py-3 border-b border-gray-100 last:border-b-0
                      ${snapshot.isDragging ? 'bg-gray-50 shadow-lg' : ''}`}
        >
          {/* Drag Handle */}
          <div {...provided.dragHandleProps} className="cursor-grab text-gray-400">
            <GripVertical className="w-4 h-4" />
          </div>
          
          {/* Icon */}
          <div className={`w-8 h-8 rounded flex items-center justify-center flex-shrink-0
                          ${resource.type === 'file' ? 'bg-red-100' : 'bg-blue-100'}`}>
            {resource.type === 'file' ? (
              <FileText className="w-4 h-4 text-red-500" />
            ) : (
              <ExternalLink className="w-4 h-4 text-blue-500" />
            )}
          </div>
          
          {/* Name */}
          <div className="flex-1 min-w-0">
            <p className="text-sm font-medium text-gray-900 truncate">{resource.name}</p>
            <p className="text-xs text-gray-400 truncate">
              {resource.type === 'file' ? resource.fileName : resource.url}
            </p>
          </div>
          
          {/* Toggle Active */}
          <label className="relative inline-flex items-center cursor-pointer">
            <input 
              type="checkbox" 
              checked={resource.isActive}
              onChange={() => onToggle(resource.id)}
              className="sr-only peer"
            />
            <div className="w-9 h-5 bg-gray-200 peer-focus:outline-none rounded-full peer 
                            peer-checked:after:translate-x-full peer-checked:after:border-white

            </div>
          </label>
          
          {/* Actions */}
          <div className="flex items-center gap-1">
            <button 
              onClick={() => onEdit(resource)}
              className="p-1.5 text-gray-400 hover:text-gray-600 rounded"
            >
              <Pencil className="w-4 h-4" />
```

## Part 2: Backend API

### Database Schema

```sql
-- Add resources table
CREATE TABLE agent_resources (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  name VARCHAR(255) NOT NULL,
  type VARCHAR(20) NOT NULL CHECK (type IN ('link', 'file')),
  url TEXT,                    -- For external links
  file_url TEXT,               -- For uploaded files (S3/storage URL)
  file_name VARCHAR(255),      -- Original file name
  is_active BOOLEAN DEFAULT true,
  display_order INTEGER DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_agent_resources_user ON agent_resources(user_id);
```

### API Endpoints

```typescript
// server/routes/resources.ts

// GET /api/resources - Get all resources for current user
router.get('/resources', async (req, res) => {
  const userId = req.user.id;
  const resources = await db.query(`
    SELECT * FROM agent_resources 
    WHERE user_id = $1 
    ORDER BY display_order ASC, created_at ASC
  `, [userId]);
  res.json(resources.rows);
});

// POST /api/resources - Create new resource
router.post('/resources', async (req, res) => {
  const userId = req.user.id;
  const { name, type, url, fileUrl, fileName } = req.body;
  
  // Get max order
  const maxOrder = await db.query(`
    SELECT COALESCE(MAX(display_order), 0) + 1 as next_order 
    FROM agent_resources WHERE user_id = $1
  `, [userId]);
  
  const result = await db.query(`
    INSERT INTO agent_resources (user_id, name, type, url, file_url, file_name, display_order)
    VALUES ($1, $2, $3, $4, $5, $6, $7)
    RETURNING *
  `, [userId, name, type, url, fileUrl, fileName, maxOrder.rows[0].next_order]);
  
  res.json(result.rows[0]);
});

// PATCH /api/resources/:id - Update resource
router.patch('/resources/:id', async (req, res) => {
  const { id } = req.params;
  const { name, url, isActive } = req.body;
  
  const result = await db.query(`
    UPDATE agent_resources 
    SET name = COALESCE($1, name),
        url = COALESCE($2, url),
        is_active = COALESCE($3, is_active),
        updated_at = CURRENT_TIMESTAMP
    WHERE id = $4 AND user_id = $5
    RETURNING *
  `, [name, url, isActive, id, req.user.id]);
  
  res.json(result.rows[0]);
});

// DELETE /api/resources/:id - Delete resource
router.delete('/resources/:id', async (req, res) => {
  await db.query(`
    DELETE FROM agent_resources WHERE id = $1 AND user_id = $2
  `, [req.params.id, req.user.id]);
  res.json({ success: true });
});

// PATCH /api/resources/reorder - Reorder resources
router.patch('/resources/reorder', async (req, res) => {
  const { orderedIds } = req.body; // Array of IDs in new order
  
  for (let i = 0; i < orderedIds.length; i++) {
    await db.query(`
      UPDATE agent_resources SET display_order = $1 WHERE id = $2 AND user_id = $3
    `, [i, orderedIds[i], req.user.id]);
  }
  
  res.json({ success: true });
});

// POST /api/resources/upload - Upload file
router.post('/resources/upload', upload.single('file'), async (req, res) => {
  // Handle file upload to S3 or local storage
  const fileUrl = await uploadToStorage(req.file);
  
  res.json({
    fileUrl,
    fileName: req.file.originalname,
  });
});
```
