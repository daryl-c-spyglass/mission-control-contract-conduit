## Fix CMA Map Runtime Error: "Cannot read properties of undefined (reading 'get')"

This error occurs when trying to access Mapbox map methods before the map is fully initialized or after it's been destroyed.

---

## Error

```
[plugin:runtime-error-plugin] Cannot read properties of undefined (reading 'get')
```

This typically happens when calling:
- `map.current.getLayer()`
- `map.current.getSource()`
- `map.current.getStyle()`
- Or any map method before initialization

---

## Root Causes

1. **Map not initialized yet** - Calling methods before `map.current` is set
2. **Map destroyed** - Calling methods after component unmounts
3. **Style not loaded** - Calling layer methods before `style.load` event
4. **Theme context undefined** - `useMapThemeContext()` called outside provider

---

## Fix 1: Add Null Checks to All Map Operations

```tsx
// components/cma-map.tsx

// BEFORE (causes error)
const existingLayer = map.current.getLayer('cma-polygon-fill');

// AFTER (safe)
const existingLayer = map.current?.getLayer?.('cma-polygon-fill');
```

---

## Fix 2: Guard All Map Operations

```tsx
// Helper function to safely execute map operations
const safeMapOperation = (operation: () => void) => {
  if (!map.current) {
    console.warn('Map not initialized');
    return;
  }
  
  try {
    operation();
  } catch (error) {
    console.error('Map operation failed:', error);
  }
};

// Usage
safeMapOperation(() => {
  if (map.current.getLayer('cma-polygon-fill')) {
    map.current.removeLayer('cma-polygon-fill');
  }
});
```

## Fix 3: Complete CMA Map Component with Safety Guards

```tsx
// components/cma-map.tsx
import { useEffect, useRef, useState, useMemo, useCallback } from 'react';
import mapboxgl from 'mapbox-gl';
import 'mapbox-gl/dist/mapbox-gl.css';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import { X, Locate, Maximize2 } from 'lucide-react';

interface CMAMapProps {
  properties: any[];
  subjectProperty?: any;
  showPolygon?: boolean;
}

export function CMAMap({ properties, subjectProperty, showPolygon = true }: CMAMapProps) {
  const mapContainer = useRef<HTMLDivElement>(null);
  const map = useRef<mapboxgl.Map | null>(null);
  const markersRef = useRef<mapboxgl.Marker[]>([]);
  const [selectedProperty, setSelectedProperty] = useState<any>(null);
  const [mapLoaded, setMapLoaded] = useState(false);
  const [mapError, setMapError] = useState<string | null>(null);

  // Check if map is ready for operations
  const isMapReady = useCallback(() => {
    return map.current !== null && mapLoaded;
  }, [mapLoaded]);

  // Safe layer removal
  const safeRemoveLayer = useCallback((layerId: string) => {
    if (!map.current) return;
    
    try {
      if (map.current.getLayer(layerId)) {
        map.current.removeLayer(layerId);
      }
    } catch (e) {
      // Layer doesn't exist, ignore
    }
  }, []);

  // Safe source removal
  const safeRemoveSource = useCallback((sourceId: string) => {
    if (!map.current) return;
    
    try {
      if (map.current.getSource(sourceId)) {
        map.current.removeSource(sourceId);
      }
    } catch (e) {
      // Source doesn't exist, ignore
    }
  }, []);

  // Calculate polygon coordinates
  const polygonCoordinates = useMemo(() => {
    if (!properties || properties.length === 0) return null;
    
    const coords: [number, number][] = [];
    
    if (subjectProperty?.map?.latitude && subjectProperty?.map?.longitude) {
      coords.push([subjectProperty.map.longitude, subjectProperty.map.latitude]);
    }
    
    properties.forEach(p => {
      if (p?.map?.latitude && p?.map?.longitude) {
        coords.push([p.map.longitude, p.map.latitude]);
      }
    });
    
    if (coords.length < 3) return null;
    
    try {
      return calculateConvexHull(coords);
    } catch (e) {
      console.error('Failed to calculate convex hull:', e);
      return null;
    }
  }, [properties, subjectProperty]);

  // Initialize map
  useEffect(() => {
    if (!mapContainer.current) return;
    
    // Prevent re-initialization
    if (map.current) return;

    const token = import.meta.env.VITE_MAPBOX_ACCESS_TOKEN;
    
    if (!token) {
      setMapError('Mapbox access token not configured');
      console.error('Mapbox access token not found');
      return;
    }

    mapboxgl.accessToken = token;

    try {
      // Calculate center
      let centerLng = -97.7431; // Austin default
      let centerLat = 30.2672;

      if (subjectProperty?.map?.longitude && subjectProperty?.map?.latitude) {
        centerLng = subjectProperty.map.longitude;
        centerLat = subjectProperty.map.latitude;
      } else if (properties?.[0]?.map?.longitude && properties?.[0]?.map?.latitude) {
        centerLng = properties[0].map.longitude;
        centerLat = properties[0].map.latitude;
      }

      map.current = new mapboxgl.Map({
        container: mapContainer.current,
        style: 'mapbox://styles/mapbox/streets-v12',
        center: [centerLng, centerLat],
        zoom: 11,
      });

      map.current.addControl(new mapboxgl.NavigationControl(), 'top-right');

      map.current.on('load', () => {
        setMapLoaded(true);
        setMapError(null);
      });

      map.current.on('error', (e) => {
        console.error('Mapbox error:', e);
        setMapError('Map failed to load');
      });

    } catch (error) {
      console.error('Failed to initialize map:', error);
      setMapError('Failed to initialize map');
    }

    // Cleanup
    return () => {
      markersRef.current.forEach(m => {
        try {
          m.remove();
        } catch (e) {
          // Marker already removed
        }
      });
      markersRef.current = [];
      
      if (map.current) {
        try {
          map.current.remove();
        } catch (e) {
          // Map already removed
        }
        map.current = null;
      }
      
      setMapLoaded(false);
    };
  }, []); // Empty deps - only run once

  // Update markers and polygon when data changes
  useEffect(() => {
    if (!isMapReady()) return;
    
    // Clear existing markers safely
    markersRef.current.forEach(m => {
      try {
        m.remove();
      } catch (e) {
        // Marker already removed
      }
    });
    markersRef.current = [];

    // Add polygon
    if (showPolygon && polygonCoordinates && polygonCoordinates.length >= 3) {
      addPolygonBoundary(polygonCoordinates);
    }

    // Add subject marker
    if (subjectProperty?.map?.latitude && subjectProperty?.map?.longitude) {
      addSubjectMarker();
    }

    // Add comparable markers
    if (properties && properties.length > 0) {
      properties.forEach(property => {
        if (property?.map?.latitude && property?.map?.longitude) {
          addPropertyMarker(property);
        }
      });
    }

    // Fit bounds
    fitBounds();
    
  }, [properties, subjectProperty, mapLoaded, polygonCoordinates, showPolygon]);

  // Add polygon boundary
  const addPolygonBoundary = useCallback((coordinates: [number, number][]) => {
    if (!map.current || !mapLoaded) return;

    try {
      // Remove existing layers first
      safeRemoveLayer('cma-polygon-fill');
      safeRemoveLayer('cma-polygon-line');
      safeRemoveSource('cma-polygon');

      // Close the polygon
      const closedCoords = [...coordinates, coordinates[0]];

      map.current.addSource('cma-polygon', {
        type: 'geojson',
        data: {
          type: 'Feature',
          properties: {},
          geometry: {
            type: 'Polygon',
            coordinates: [closedCoords],
          },
        },
      });

      map.current.addLayer({
        id: 'cma-polygon-fill',
        type: 'fill',
        source: 'cma-polygon',
        paint: {
          'fill-color': '#3b82f6',
          'fill-opacity': 0.15,
        },
      });

      map.current.addLayer({
        id: 'cma-polygon-line',
        type: 'line',
        source: 'cma-polygon',
        paint: {
          'line-color': '#2563eb',
          'line-width': 3,
          'line-opacity': 0.8,
        },
      });
    } catch (error) {
      console.error('Failed to add polygon:', error);
    }
  }, [mapLoaded, safeRemoveLayer, safeRemoveSource]);

  // Add subject marker
  const addSubjectMarker = useCallback(() => {
    if (!map.current || !subjectProperty?.map) return;

    const { latitude, longitude } = subjectProperty.map;
    if (!latitude || !longitude) return;

    try {
      const price = subjectProperty.listPrice || 0;
      const priceLabel = formatPrice(price);

      const el = document.createElement('div');
      el.innerHTML = `
        <div style="
          background: linear-gradient(135deg, #2563eb 0%, #1e40af 100%);
          color: white;
          padding: 6px 12px;
          border-radius: 6px;
          font-family: system-ui, sans-serif;
          font-weight: 700;
          font-size: 13px;
          box-shadow: 0 4px 12px rgba(37, 99, 235, 0.5);
          white-space: nowrap;
          border: 2px solid white;
          cursor: pointer;
        ">
          <div style="font-size: 9px; text-transform: uppercase; opacity: 0.9;">Subject</div>
          ${priceLabel}
        </div>
      `;

      el.addEventListener('click', () => setSelectedProperty({ ...subjectProperty, isSubject: true }));

      const marker = new mapboxgl.Marker({ element: el, anchor: 'bottom' })
        .setLngLat([longitude, latitude])
        .addTo(map.current);

      markersRef.current.push(marker);
    } catch (error) {
      console.error('Failed to add subject marker:', error);
    }
  }, [subjectProperty]);

  // Add property marker
  const addPropertyMarker = useCallback((property: any) => {
    if (!map.current || !property?.map) return;

    const { latitude, longitude } = property.map;
    if (!latitude || !longitude) return;

    try {
      const status = (property.status || '').toLowerCase();
      const isSold = status.includes('sold') || status.includes('closed');
      const price = isSold
        ? (property.soldPrice || property.closePrice || property.listPrice || 0)
        : (property.listPrice || 0);
      const priceLabel = formatPrice(price);
      const bgColor = isSold ? '#dc2626' : '#16a34a';

      const el = document.createElement('div');
      el.innerHTML = `
        <div style="
          background: ${bgColor};
          color: white;
          padding: 4px 8px;
          border-radius: 4px;
          font-family: system-ui, sans-serif;
          font-weight: 600;
          font-size: 12px;
          box-shadow: 0 2px 8px rgba(0,0,0,0.2);
          white-space: nowrap;
          cursor: pointer;
        ">
          ${priceLabel}
        </div>
      `;

      el.addEventListener('click', () => setSelectedProperty(property));

      const marker = new mapboxgl.Marker({ element: el, anchor: 'bottom' })
        .setLngLat([longitude, latitude])
        .addTo(map.current);

      markersRef.current.push(marker);
    } catch (error) {
      console.error('Failed to add property marker:', error);
    }
  }, []);

  // Fit bounds
  const fitBounds = useCallback(() => {
    if (!map.current || !mapLoaded) return;

    try {
      const coords: [number, number][] = [];

      if (subjectProperty?.map?.longitude && subjectProperty?.map?.latitude) {
        coords.push([subjectProperty.map.longitude, subjectProperty.map.latitude]);
      }

      properties?.forEach(p => {
        if (p?.map?.longitude && p?.map?.latitude) {
          coords.push([p.map.longitude, p.map.latitude]);
        }
      });

      if (coords.length === 0) return;

      if (coords.length === 1) {
        map.current.setCenter(coords[0]);
        map.current.setZoom(14);
        return;
      }

      const bounds = coords.reduce(
        (bounds, coord) => bounds.extend(coord),
        new mapboxgl.LngLatBounds(coords[0], coords[0])
      );

      map.current.fitBounds(bounds, {
        padding: 60,
        maxZoom: 14,
        duration: 1000,
      });
    } catch (error) {
      console.error('Failed to fit bounds:', error);
    }
  }, [properties, subjectProperty, mapLoaded]);

  // Center on subject
  const centerOnSubject = useCallback(() => {
    if (!map.current || !subjectProperty?.map) return;
    
    try {
      map.current.flyTo({
        center: [subjectProperty.map.longitude, subjectProperty.map.latitude],
        zoom: 14,
      });
    } catch (error) {
      console.error('Failed to center on subject:', error);
    }
  }, [subjectProperty]);

  // Error state
  if (mapError) {
    return (
      <div className="w-full h-[550px] rounded-lg border bg-gray-100 flex items-center justify-center">
        <div className="text-center">
          <p className="text-red-500 font-medium">{mapError}</p>
          <p className="text-sm text-muted-foreground mt-2">Please check your Mapbox configuration</p>
        </div>
      </div>
    );
  }

  return (
    <div className="relative w-full h-[550px] rounded-lg overflow-hidden border bg-gray-100">
      <div ref={mapContainer} className="w-full h-full" />

      {/* Controls */}
      <div className="absolute top-4 left-4 flex flex-col gap-2">
        <Button
          size="icon"
          variant="secondary"
          className="h-8 w-8 shadow-md"
          onClick={centerOnSubject}
          disabled={!mapLoaded}
        >
          <Locate className="h-4 w-4" />
        </Button>
        <Button
          size="icon"
          variant="secondary"
          className="h-8 w-8 shadow-md"
          onClick={fitBounds}
          disabled={!mapLoaded}
        >
          <Maximize2 className="h-4 w-4" />
        </Button>
      </div>

      {/* Property count */}
      <div className="absolute top-4 left-1/2 -translate-x-1/2">
        <Badge variant="secondary" className="shadow-md bg-white">
          {properties?.length || 0} comparables + Subject
        </Badge>
      </div>

      {/* Legend */}
      <div className="absolute bottom-4 left-4 bg-white/95 backdrop-blur rounded-lg p-3 shadow-lg">
        <p className="text-xs font-semibold mb-2">Legend</p>
        <div className="space-y-1.5">
          <div className="flex items-center gap-2">
            <div className="w-4 h-3 rounded bg-blue-600"></div>
            <span className="text-xs">Subject</span>
          </div>
          <div className="flex items-center gap-2">
            <div className="w-4 h-3 rounded bg-green-600"></div>
            <span className="text-xs">Active</span>
          </div>
          <div className="flex items-center gap-2">
            <div className="w-4 h-3 rounded bg-red-600"></div>
            <span className="text-xs">Sold</span>
          </div>
        </div>
      </div>

      {/* Selected property popup */}
      {selectedProperty && (
        <PropertyPopup
          property={selectedProperty}
          onClose={() => setSelectedProperty(null)}
        />
      )}

      {/* Loading */}
      {!mapLoaded && !mapError && (
        <div className="absolute inset-0 flex items-center justify-center bg-gray-100/80">
          <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-blue-600"></div>
        </div>
      )}
    </div>
  );
}

// Property popup component
function PropertyPopup({ property, onClose }: { property: any; onClose: () => void }) {
  const status = (property.status || '').toLowerCase();
  const isSold = status.includes('sold') || status.includes('closed');
  const isSubject = property.isSubject;
  const price = isSold
    ? (property.soldPrice || property.closePrice || property.listPrice || 0)
    : (property.listPrice || 0);

  return (
    <div className="absolute top-4 right-4 w-72 bg-white rounded-xl shadow-2xl overflow-hidden">
      <button
        className="absolute top-2 right-2 z-10 p-1.5 bg-white/90 hover:bg-white rounded-full shadow"
        onClick={onClose}
      >
        <X className="h-4 w-4" />
      </button>

      {property.images?.[0] && (
        <img src={property.images[0]} alt="Property" className="w-full h-28 object-cover" />
      )}

      <div className="p-3">
        <Badge className={isSubject ? 'bg-blue-600' : isSold ? 'bg-red-500' : 'bg-green-500'}>
          {isSubject ? 'Subject' : isSold ? 'Sold' : 'Active'}
        </Badge>
        <p className="font-bold text-lg mt-2">${price.toLocaleString()}</p>
        <p className="text-sm text-gray-600 truncate">
          {property.address?.streetAddress || property.unparsedAddress}
        </p>
        <div className="flex gap-3 text-xs mt-2 text-muted-foreground">
          <span>{property.numBedrooms || '—'} bd</span>
          <span>{property.numBathrooms || '—'} ba</span>
          <span>{property.sqft?.toLocaleString() || '—'} sqft</span>
        </div>
      </div>
    </div>
  );
}

// Format price helper
function formatPrice(price: number): string {
  if (price >= 1000000) {
    return `$${(price / 1000000).toFixed(2)}M`;
  }
  return `$${Math.round(price / 1000)}K`;
}

// Convex hull calculation
function calculateConvexHull(points: [number, number][]): [number, number][] {
  if (!points || points.length < 3) return points || [];

  try {
    let start = 0;
    for (let i = 1; i < points.length; i++) {
      if (points[i][1] < points[start][1] ||
          (points[i][1] === points[start][1] && points[i][0] < points[start][0])) {
        start = i;
      }
    }

    const pointsCopy = [...points];
    [pointsCopy[0], pointsCopy[start]] = [pointsCopy[start], pointsCopy[0]];
    const pivot = pointsCopy[0];

    const sorted = pointsCopy.slice(1).sort((a, b) => {
      const angleA = Math.atan2(a[1] - pivot[1], a[0] - pivot[0]);
      const angleB = Math.atan2(b[1] - pivot[1], b[0] - pivot[0]);
      return angleA - angleB;
    });

    const hull: [number, number][] = [pivot];
    for (const point of sorted) {
      while (hull.length > 1) {
        const cross = (hull[hull.length - 1][0] - hull[hull.length - 2][0]) * (point[1] - hull[hull.length - 2][1]) -
                      (hull[hull.length - 1][1] - hull[hull.length - 2][1]) * (point[0] - hull[hull.length - 2][0]);
        if (cross <= 0) hull.pop();
        else break;
      }
      hull.push(point);
    }

    return hull;
  } catch (error) {
    console.error('Convex hull calculation error:', error);
    return points;
  }
}
```

## Fix 4: Check Theme Context Provider

If using the map theme context, ensure the provider wraps the component:

```tsx
// Make sure MapThemeProvider exists in parent
// If not using theme context, remove the import

// OPTION A: Use the context safely
import { useMapThemeContext } from '@/contexts/map-theme-context';

function CMAMap() {
  // This will throw if provider is missing
  const { mapStyle } = useMapThemeContext();
}

// OPTION B: Use a default if context is missing
function useMapThemeSafe() {
  try {
    return useMapThemeContext();
  } catch {
    return {
      mapStyle: 'mapbox://styles/mapbox/streets-v12',
      isDarkMap: false,
    };
  }
}
```

## Summary of Fixes

| Issue | Fix |
|-------|-----|
| `map.current` is null | Add null checks: `map.current?.getLayer()` |
| Layer doesn't exist | Use `safeRemoveLayer()` helper |
| Source doesn't exist | Use `safeRemoveSource()` helper |
| Style not loaded | Check `mapLoaded` state before operations |
| Theme context missing | Ensure `MapThemeProvider` wraps app or use safe default |
| Cleanup errors | Wrap cleanup in try-catch |
