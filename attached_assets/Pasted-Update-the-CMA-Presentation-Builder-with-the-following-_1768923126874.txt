### State for Layout Tab:
```tsx
// Add these state variables
const [photosPerProperty, setPhotosPerProperty] = useState('2'); // Default: 2 photos
const [photoSelection, setPhotoSelection] = useState('first-12'); // Default: First 12
const [selectedPhotos, setSelectedPhotos] = useState<number[]>([]); // For custom selection
const [propertyPhotos, setPropertyPhotos] = useState<string[]>([]); // From Repliers API

// Toggle photo selection for custom mode
const togglePhotoSelection = (index: number) => {
  setSelectedPhotos(prev => 
    prev.includes(index) 
      ? prev.filter(i => i !== index)
      : [...prev, index]
  );
};

// Fetch photos from Repliers API (should already be available from MLS data)
useEffect(() => {
  if (mlsData?.photos) {
    setPropertyPhotos(mlsData.photos);
  }
}, [mlsData]);
```

## PHOTO SELECTION LOGIC FOR PDF EXPORT

When generating the PDF, use the layout settings to determine which photos to include:
```tsx
// In PDF generation logic
const getPhotosForReport = (photos: string[], settings: LayoutSettings) => {
  const { photoSelection, photosPerProperty, selectedPhotos } = settings;
  
  let selectedPhotoUrls: string[] = [];
  
  switch (photoSelection) {
    case 'first-12':
      selectedPhotoUrls = photos.slice(0, 12);
      break;
    case 'all':
      selectedPhotoUrls = photos;
      break;
    case 'ai-suggest':
      // Use Repliers Image Insights API to get best quality photos
      // Or fall back to first N photos
      selectedPhotoUrls = photos.slice(0, 12); // Placeholder
      break;
    case 'custom':
      selectedPhotoUrls = selectedPhotos.map(index => photos[index]).filter(Boolean);
      break;
    default:
      selectedPhotoUrls = photos.slice(0, 12);
  }
  
  // Limit to photos per property setting
  const photosPerProp = parseInt(photosPerProperty) || 2;
  
  return {
    photos: selectedPhotoUrls,
    photosPerProperty: photosPerProp,
  };
};
```
