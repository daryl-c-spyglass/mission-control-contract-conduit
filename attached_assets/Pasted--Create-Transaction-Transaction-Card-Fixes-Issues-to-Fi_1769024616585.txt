### 4b. Add Archive Button to Transaction Detail View

**File: Update your transaction detail/card component**

```tsx
import { Archive, ArchiveRestore, Trash2 } from 'lucide-react';
import { useMutation, useQueryClient } from '@tanstack/react-query';
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from '@/components/ui/alert-dialog';
import {
  DropdownMenu,
  DropdownMenuContent,
  DropdownMenuItem,
  DropdownMenuSeparator,
  DropdownMenuTrigger,
} from '@/components/ui/dropdown-menu';
import { MoreVertical } from 'lucide-react';

interface TransactionActionsProps {
  transaction: {
    id: number;
    address: string;
    isArchived?: boolean;
  };
}

export function TransactionActions({ transaction }: TransactionActionsProps) {
  const queryClient = useQueryClient();

  // Archive mutation
  const archiveTransaction = useMutation({
    mutationFn: async (id: number) => {
      const res = await fetch(`/api/transactions/${id}/archive`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ isArchived: true }),
      });
      if (!res.ok) throw new Error('Failed to archive');
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['transactions'] });
      queryClient.invalidateQueries({ queryKey: ['archived-transactions'] });
      toast({ title: 'Transaction archived' });
    },
  });

  // Unarchive mutation
  const unarchiveTransaction = useMutation({
    mutationFn: async (id: number) => {
      const res = await fetch(`/api/transactions/${id}/archive`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ isArchived: false }),
      });
      if (!res.ok) throw new Error('Failed to unarchive');
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['transactions'] });
      queryClient.invalidateQueries({ queryKey: ['archived-transactions'] });
      toast({ title: 'Transaction restored' });
    },
  });

  // Delete mutation
  const deleteTransaction = useMutation({
    mutationFn: async (id: number) => {
      const res = await fetch(`/api/transactions/${id}`, { method: 'DELETE' });
      if (!res.ok) throw new Error('Failed to delete');
      return res.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['transactions'] });
      queryClient.invalidateQueries({ queryKey: ['archived-transactions'] });
      toast({ title: 'Transaction deleted' });
    },
  });

  return (
    <DropdownMenu>
      <DropdownMenuTrigger asChild>
        <Button variant="ghost" size="sm">
          <MoreVertical className="w-4 h-4" />
        </Button>
      </DropdownMenuTrigger>
      <DropdownMenuContent align="end">
        {/* Archive / Unarchive */}
        {transaction.isArchived ? (
          <DropdownMenuItem
            onClick={() => unarchiveTransaction.mutate(transaction.id)}
            className="cursor-pointer"
          >
            <ArchiveRestore className="w-4 h-4 mr-2" />
            Restore from Archive
          </DropdownMenuItem>
        ) : (
          <DropdownMenuItem
            onClick={() => archiveTransaction.mutate(transaction.id)}
            className="cursor-pointer"
          >
            <Archive className="w-4 h-4 mr-2" />
            Archive Transaction
          </DropdownMenuItem>
        )}

        <DropdownMenuSeparator />

        {/* Delete with confirmation */}
        <AlertDialog>
          <AlertDialogTrigger asChild>
            <DropdownMenuItem
              onSelect={(e) => e.preventDefault()}
              className="text-red-600 cursor-pointer"
            >
              <Trash2 className="w-4 h-4 mr-2" />
              Delete Transaction
            </DropdownMenuItem>
          </AlertDialogTrigger>
          <AlertDialogContent>
            <AlertDialogHeader>
              <AlertDialogTitle>Delete Transaction</AlertDialogTitle>
              <AlertDialogDescription>
                Are you sure you want to permanently delete "{transaction.address}"?
                This action cannot be undone.
              </AlertDialogDescription>
            </AlertDialogHeader>
            <AlertDialogFooter>
              <AlertDialogCancel>Cancel</AlertDialogCancel>
              <AlertDialogAction
                onClick={() => deleteTransaction.mutate(transaction.id)}
                className="bg-red-500 hover:bg-red-600"
              >
                Delete
              </AlertDialogAction>
            </AlertDialogFooter>
          </AlertDialogContent>
        </AlertDialog>
      </DropdownMenuContent>
    </DropdownMenu>
  );
}
```

### 4c. Backend Routes for Archive

**File: `server/routes.ts`**

```typescript
// Archive/Unarchive transaction
app.patch('/api/transactions/:id/archive', async (req, res) => {
  try {
    const { id } = req.params;
    const { isArchived } = req.body;

    const [updated] = await db
      .update(transactions)
      .set({
        isArchived,
        archivedAt: isArchived ? new Date() : null,
        updatedAt: new Date(),
      })
      .where(eq(transactions.id, parseInt(id)))
      .returning();

    res.json(updated);
  } catch (error: any) {
    console.error('Archive transaction error:', error);
    res.status(500).json({ error: error.message });
  }
});

// Get archived transactions
app.get('/api/transactions/archived', async (req, res) => {
  try {
    const userId = req.user?.id; // If using auth
    
    const archivedTransactions = await db
      .select()
      .from(transactions)
      .where(
        and(
          eq(transactions.isArchived, true),
          userId ? eq(transactions.userId, userId) : undefined
        )
      )
      .orderBy(desc(transactions.archivedAt));

    res.json(archivedTransactions);
  } catch (error: any) {
    console.error('Get archived transactions error:', error);
    res.status(500).json({ error: error.message });
  }
});

// Update existing GET /api/transactions to exclude archived
app.get('/api/transactions', async (req, res) => {
  try {
    const userId = req.user?.id;
    
    const activeTransactions = await db
      .select()
      .from(transactions)
      .where(
        and(
          eq(transactions.isArchived, false), // â† ADD THIS FILTER
          userId ? eq(transactions.userId, userId) : undefined
        )
      )
      .orderBy(desc(transactions.createdAt));

    res.json(activeTransactions);
  } catch (error: any) {
    console.error('Get transactions error:', error);
    res.status(500).json({ error: error.message });
  }
});
```

### 4d. Archive Page Component

**File: `client/src/pages/archive.tsx`** (CREATE)

```tsx
import React from 'react';
import { useQuery } from '@tanstack/react-query';
import { TransactionCard } from '@/components/transactions/transaction-card';
import { Archive as ArchiveIcon } from 'lucide-react';

export default function ArchivePage() {
  const { data: archivedTransactions, isLoading } = useQuery({
    queryKey: ['archived-transactions'],
    queryFn: async () => {
      const res = await fetch('/api/transactions/archived');
      if (!res.ok) throw new Error('Failed to fetch archived transactions');
      return res.json();
    },
  });

  if (isLoading) {
    return (
      <div className="flex items-center justify-center h-64">
        <div className="animate-spin rounded-full h-8 w-8 border-b-2 border-primary" />
      </div>
    );
  }

  return (
    <div className="space-y-6">
      <div>
        <h1 className="text-2xl font-bold">Archive</h1>
        <p className="text-muted-foreground">
          Archived transactions that are no longer active
        </p>
      </div>

      {archivedTransactions?.length === 0 ? (
        <div className="flex flex-col items-center justify-center h-64 text-center">
          <ArchiveIcon className="w-12 h-12 text-muted-foreground mb-4" />
          <h3 className="font-medium text-lg">No Archived Transactions</h3>
          <p className="text-muted-foreground text-sm mt-1">
            Archived transactions will appear here
          </p>
        </div>
      ) : (
        <div className="grid gap-4 md:grid-cols-2 lg:grid-cols-3">
          {archivedTransactions?.map((transaction: any) => (
            <TransactionCard
              key={transaction.id}
              transaction={transaction}
              showArchiveActions
            />
          ))}
        </div>
      )}
    </div>
  );
}
```
